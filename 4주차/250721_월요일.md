-- 1.FROM > 2.WHERE > 3.GROUP BY > 4.HAVING > 5.SELECT > 6.ORDER BY > 7. LIMIT 
-- group by: 원본은 그대로 두고, 원본테이블에 없는 값을 계산해 냅니다 
-- 각 연도별 만들어진 영화의 총 편수를 각각 집계
SELECT 
    YEAR(release_date), COUNT(release_date)
FROM
    movies
WHERE
    release_date IS NOT NULL
GROUP BY YEAR(release_date)
HAVING COUNT(release_date) >= 300 -- GROUP BY의 결과에 조건을 또 걸때
ORDER BY YEAR(release_date) DESC;
-- group by의 결과에서 조건을 만족하는 값만 조회 
-- 각 연도별 최대 관객 수
-- groupby한 결과는 기준대로 새롭게 만들어진 테이블입니다. 여기에서 원본 테이블의 한 행의 정보를 확인하는 것은 기본적으로는 불가능합니다.
-- 특정 결과를 행 단위로 보고 싶다면 join 해서 기존 테이블과 확인합니다.
SELECT 
	-- title,
    YEAR(release_date) AS 총년도, SUM(audience) AS 총관객수, MAX(audience) AS 최대관객수
FROM
    movies
WHERE
    release_date IS NOT NULL
GROUP BY YEAR(release_date)
HAVING 총관객수 > 0
ORDER BY SUM(audience) DESC;

/* 1. 2019년 1분기(QUARTER 함수 사용) 개봉 영화 중

 월별, 영화 유형별 관객 소계를 구하는 쿼리 */

SELECT MONTH(release_date) AS 월, movie_type AS `영화 유형`,SUM(audience) AS `관객 소계` FROM movies WHERE YEAR(release_date) = 2019 AND QUARTER(release_date) = 1 AND audience >= 100000 GROUP BY 월, `영화 유형` ORDER BY MONTH(release_date) ASC;

/* 2. 2019년 개봉 영화 중 매출액이 천만 원 이상인 영화의 월별(MONTH), 
	영화 유형별 관객 소계를 구하되
	7월 1일 전에 개봉한 영화이면 상반기,
	7월 1일 이후에 개봉한 영화이면 하반기라고 함께 출력하는 쿼리 */
SELECT MONTH(release_date) AS `월`, movie_type AS `영화 유형`, SUM(audience) AS `관객 소계`, IF(MONTH(release_date) <= 6,'상반기','하반기') AS `반기` FROM movies WHERE YEAR(release_date) = 2019 AND revenue >= 10000000 GROUP BY `월`, `영화 유형`, `반기`;

/* 3. 부제가 있는 영화 찾기 ':' 2015년 이후의 개봉영화 중에서

SELECT * FROM movies WHERE YEAR(release_date) >= 2015 AND title LIKE '%:%';

/* 4. 감독이 두 명 이상이면 슬래시(/)로 이어져 있습니다(예, ‘홍길동/김감독’). 감독이 1명이면 그대로,

SELECT REPLACE(director,'/',',') AS 감독 FROM movies WHERE director IS NOT NULL;
